#include <stdio.h>

/*
Desafio MateCheck - Nível Novato
--------------------------------
Movimentação de peças usando estruturas de repetição

Peças e movimentos:
- Bispo: 5 casas na diagonal superior direita (Cima + Direita)
- Torre: 5 casas para a direita
- Rainha: 8 casas para a esquerda

Autor: [Seu Nome]
Data: [Data de entrega]
*/

int main() {
    // Constantes definindo a quantidade de movimentos de cada peça
    const int BISPO_MOV = 5;   // 5 casas
    const int TORRE_MOV = 5;   // 5 casas
    const int RAINHA_MOV = 8;  // 8 casas

    int i;

    // ------------------ BISPO ------------------
    printf("=== Movimento do Bispo ===\n");
    for (i = 1; i <= BISPO_MOV; i++) {
        printf("Cima\n");
        printf("Direita\n");
    }
    printf("\n");

    // ------------------ TORRE ------------------
    printf("=== Movimento da Torre ===\n");
    i = 1;
    while (i <= TORRE_MOV) {
        printf("Direita\n");
        i++;
    }
    printf("\n");

    // ------------------ RAINHA ------------------
    printf("=== Movimento da Rainha ===\n");
    i = 1;
    do {
        printf("Esquerda\n");
        i++;
    } while (i <= RAINHA_MOV);
    printf("\n");

    return 0;
}
